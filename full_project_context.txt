from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
import redis
import os

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

REDIS_HOST = os.getenv("REDIS_HOST", "pos_redis")
REDIS_PORT = int(os.getenv("REDIS_PORT", 6379))

# decode_responses=True –ø–æ–≤–µ—Ä—Ç–∞—î —Å—Ç—Ä–æ–∫–∏
r = redis.Redis(host=REDIS_HOST, port=REDIS_PORT, db=0, decode_responses=True)
CART_KEY = "cart"

@app.get("/")
def read_root():
    return {"message": "Order Service is running!"}

# --- –í–ò–ü–†–ê–í–õ–ï–ù–ê –§–£–ù–ö–¶–Ü–Ø ---
@app.get("/cart/")
def get_cart():
    raw_cart = r.hgetall(CART_KEY)
    
    # –ú–∏ —Ñ–æ—Ä–º—É—î–º–æ –°–ª–æ–≤–Ω–∏–∫ (Dictionary), –∞ –Ω–µ —Å–ø–∏—Å–æ–∫
    # –ë—É–ª–æ: {"item:9": "2"} -> –°—Ç–∞–Ω–µ: {"9": 2}
    clean_cart = {}
    
    for key, value in raw_cart.items():
        try:
            # key = "item:9" -> parts = ["item", "9"]
            parts = key.split(":")
            if len(parts) == 2 and parts[0] == "item":
                item_id = parts[1] # –ó–∞–ª–∏—à–∞—î–º–æ —è–∫ —Å—Ç—Ä—ñ—á–∫—É "9", –±–æ JSON –∫–ª—é—á—ñ –∑–∞–≤–∂–¥–∏ —Å—Ç—Ä—ñ—á–∫–∏
                quantity = int(value)
                
                clean_cart[item_id] = quantity
        except ValueError:
            continue

    # –†–µ–∑—É–ª—å—Ç–∞—Ç: {"9": 2, "14": 1}
    return clean_cart

@app.post("/cart/{item_id}")
def add_item(item_id: int):
    r.hincrby(CART_KEY, f"item:{item_id}", 1)
    return {"status": "added", "id": item_id}

@app.post("/cart/{item_id}/decrease")
def decrease_item(item_id: int):
    key = f"item:{item_id}"
    new_qty = r.hincrby(CART_KEY, key, -1)
    if new_qty <= 0:
        r.hdel(CART_KEY, key)
    return {"status": "decreased", "id": item_id, "qty": new_qty}

@app.delete("/cart/{item_id}")
def remove_item(item_id: int):
    key = f"item:{item_id}"
    r.hdel(CART_KEY, key)
    return {"status": "removed", "id": item_id}

@app.delete("/cart/")
def clear_cart():
    r.delete(CART_KEY)
    return {"status": "cart cleared"}

            for r_item in v.recipe:

                    variant_id=db_variant.id,

                    quantity=r_item.quantity
                )
                db.add(db_var_recipe)

    # 4. –ú–æ–¥–∏—Ñ—ñ–∫–∞—Ç–æ—Ä–∏
    for group in product.modifier_groups:

    db.refresh(new_order)

        if not product: continue

        item_name = product.name
        price = product.price
        details_list = []

        # –ê. –Ø–∫—â–æ —Ü–µ –í–ê–†–Ü–ê–ù–¢
        if item.variant_id:
            variant = db.query(models.ProductVariant).filter(models.ProductVariant.id == item.variant_id).first()
            if variant:

                price = variant.price
                details_list.append(f"–í–∞—Ä—ñ–∞–Ω—Ç: {variant.name}")
                
                # –°–ø–∏—Å–∞–Ω–Ω—è –ø–æ —Ä–µ—Ü–µ–ø—Ç—É –≤–∞—Ä—ñ–∞–Ω—Ç—É
                for r in variant.variant_recipe:
                     if r.ingredient:
                        r.ingredient.stock_quantity -= r.quantity * item.quantity
                        db.add(r.ingredient)
        else:
            # –°–ø–∏—Å–∞–Ω–Ω—è –ø–æ –ø—Ä–æ—Å—Ç–æ–º—É —Ä–µ—Ü–µ–ø—Ç—É
             for r in product.recipe:
                 if r.ingredient:
                    r.ingredient.stock_quantity -= r.quantity * item.quantity
                    db.add(r.ingredient)

        # –ë. –û–±—Ä–æ–±–∫–∞ –ú–û–î–ò–§–Ü–ö–ê–¢–û–†–Ü–í
        for mod_item in item.modifiers:
            mod = db.query(models.Modifier).filter(models.Modifier.id == mod_item.modifier_id).first()
            if mod:
                details_list.append(mod.name)
                # –Ø–∫—â–æ –º–æ–¥–∏—Ñ—ñ–∫–∞—Ç–æ—Ä —â–æ—Å—å —Å–ø–∏—Å—É—î (–Ω–∞–ø—Ä. –∫–æ—Ä–æ–±–∫—É)

        order_item = models.OrderItem(
            order_id=new_order.id,
            product_name=item_name,
